var express = require('express')  , app = express() , fs = require('fs');  // Routes  app.get('/form', function(req, res)   {   fs.readFile('./form.html', function(error, content)     { if (error)        { res.writeHead(500); res.end();      }     else     { res.writeHead(200, { 'Content-Type': 'text/html' }); res.end(content, 'utf-8');     }    })  }).listen(process.env.PORT);
